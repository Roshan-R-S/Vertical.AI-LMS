Database Schema

4 core entities with clear relationships. Activities serve as the audit trail for each lead's pipeline progress.

Design decision: Each Activity record is an immutable log entry (one row per interaction). This gives you full history of every call, meeting, and status change — critical for CRM analytics.

Prisma Schema

Copy this into prisma/schema.prisma. Run npx prisma migrate dev to generate migrations.

// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ──────────────────────────── ENUMS ────────────────────────────

enum Role {
  ADMIN
  EXECUTIVE
}

enum Milestone {
  FIRST_CALL
  SECOND_CALL
  THIRD_CALL
  EMAIL_THE_DETAILS
  MEETING_SCHEDULED
  MEETING_POSTPONED
  MEETING_EXECUTED
  SECOND_LEVEL_MEETING
  NEGOTIATION
}

enum Submilestone {
  INTERESTED
  FOLLOW_UP
  NOT_RIGHT_SPOC
  DNP          // Did Not Pick
  DECLINED
  SWITCHED_OFF
  INVALID_NUMBER
  NOT_INTERESTED
  BUSY
}

// ──────────────────────────── MODELS ───────────────────────────

model User {
  id           String     @id @default(cuid())
  name         String
  email        String     @unique
  passwordHash String
  role         Role       @default(EXECUTIVE)
  leads        Lead[]     // leads assigned to this exec
  activities   Activity[] // activities logged by this exec
  createdAt    DateTime   @default(now())
  updatedAt    DateTime   @updatedAt

  @@map("users")
}

model Company {
  id        String   @id @default(cuid())
  name      String   @unique
  industry  String?
  website   String?
  leads     Lead[]
  createdAt DateTime @default(now())

  @@map("companies")
}

model Lead {
  id                  String       @id @default(cuid())
  name                String
  email               String?
  mobile              String
  designation         String?
  currentMilestone    Milestone   @default(FIRST_CALL)
  currentSubmilestone Submilestone @default(FOLLOW_UP)
  company             Company     @relation(fields: [companyId], references: [id])
  companyId           String
  assignedTo          User        @relation(fields: [assignedToId], references: [id])
  assignedToId        String
  activities          Activity[]
  createdAt           DateTime    @default(now())
  updatedAt           DateTime    @updatedAt

  @@index([companyId])
  @@index([assignedToId])
  @@index([currentMilestone])
  @@map("leads")
}

model Activity {
  id           String       @id @default(cuid())
  lead         Lead        @relation(fields: [leadId], references: [id], onDelete: Cascade)
  leadId       String
  executive    User        @relation(fields: [executiveId], references: [id])
  executiveId  String
  milestone    Milestone
  submilestone Submilestone
  remarks      String?
  createdAt    DateTime    @default(now())

  @@index([leadId])
  @@index([executiveId])
  @@map("activities")
}

modules/leads/leads.repository.ts
import { prisma } from "../../prisma";
import { Milestone, Submilestone } from "@prisma/client";

export const LeadRepo = {
  findAll: (filters: {
    milestone?: Milestone;
    assignedToId?: string;
    search?: string;
    skip?: number;
    take?: number;
  }) => {
    const { milestone, assignedToId, search, skip = 0, take = 20 } = filters;
    return prisma.lead.findMany({
      where: {
        ...(milestone && { currentMilestone: milestone }),
        ...(assignedToId && { assignedToId }),
        ...(search && {
          OR: [
            { name: { contains: search, mode: "insensitive" } },
            { email: { contains: search, mode: "insensitive" } },
            { company: { name: { contains: search, mode: "insensitive" } } },
          ],
        }),
      },
      include: { company: true, assignedTo: { select: { id: true, name: true } } },
      orderBy: { updatedAt: "desc" },
      skip,
      take,
    });
  },

  logActivity: async (
    leadId: string,
    executiveId: string,
    milestone: Milestone,
    submilestone: Submilestone,
    remarks?: string
  ) => {
    return prisma.$transaction([
      prisma.activity.create({
        data: { leadId, executiveId, milestone, submilestone, remarks },
      }),
      prisma.lead.update({
        where: { id: leadId },
        data: { currentMilestone: milestone, currentSubmilestone: submilestone },
      }),
    ]);
  },
};
modules/analytics/analytics.service.ts
import { prisma } from "../../prisma";

export const getPipelineBreakdown = async (executiveId?: string) => {
  return prisma.lead.groupBy({
    by: ["currentMilestone"],
    _count: { _all: true },
    where: executiveId ? { assignedToId: executiveId } : {},
    orderBy: { _count: { currentMilestone: "desc" } },
  });
};

export const getExecutiveStats = async () => {
  return prisma.user.findMany({
    select: {
      id: true,
      name: true,
      _count: { select: { leads: true, activities: true } },
    },
  });
};
prisma/seed.ts — import your Excel data
import * as xlsx from "xlsx";
import { prisma } from "../src/prisma";
import { Milestone, Submilestone } from "@prisma/client";

const MILESTONE_MAP: Record<string, Milestone> = {
  "First Call":           "FIRST_CALL",
  "Second Call":          "SECOND_CALL",
  "Third Call":           "THIRD_CALL",
  "Email the Details":    "EMAIL_THE_DETAILS",
  "Meeting Scheduled":    "MEETING_SCHEDULED",
  "Meeting Postponed":    "MEETING_POSTPONED",
  "Meeting Executed":     "MEETING_EXECUTED",
  "2nd Level Meeting":    "SECOND_LEVEL_MEETING",
  "Negotiation":          "NEGOTIATION",
};

async function main() {
  const wb = xlsx.readFile("./data/leads.xlsx");
  const rows = xlsx.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]]);

  for (const row of rows) {
    const r = row as any;
    const company = await prisma.company.upsert({
      where: { name: r.Company },
      create: { name: r.Company },
      update: {},
    });

    // upsert executive by name
    const exec = await prisma.user.upsert({
      where: { email: `${r.Exe.toLowerCase()}@crm.internal` },
      create: { name: r.Exe, email: `${r.Exe.toLowerCase()}@crm.internal`, passwordHash: "" },
      update: {},
    });

    const lead = await prisma.lead.create({
      data: {
        name:               r.Name,
        email:              r.Email || null,
        mobile:             String(r["Mobile No"]),
        designation:        r.Designation || null,
        companyId:          company.id,
        assignedToId:       exec.id,
        currentMilestone:   MILESTONE_MAP[r.Milestone] ?? "FIRST_CALL",
        currentSubmilestone: "FOLLOW_UP",
      },
    });

    await prisma.activity.create({
      data: {
        leadId:      lead.id,
        executiveId: exec.id,
        milestone:   lead.currentMilestone,
        submilestone: "FOLLOW_UP",
        remarks:     r.Remarks || null,
      },
    });
  }
  console.log("✓ Seed complete");
}

main().catch(console.error).finally(() => prisma.$disconnect());
